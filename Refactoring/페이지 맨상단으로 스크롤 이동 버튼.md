### JavaScript

```javascript
initTopBtn : function(){
	const welNaviTop = documents.querySelector('#_naviToTop');
	const htmlTag = documents.getElementsByTagName('html');

	welNaviTop.addEventListener('click', function(e){
		e.preventDefault();
		htmlTag.scrollTop(0);
	});
},
```
